function doPost(e) {
  // Receive data from hidden form submission (e.parameter.data)
  // or direct JSON post (e.postData.contents) — handle both
  var raw = '';
  try {
    raw = e.parameter.data || e.postData.contents || '';
  } catch(err) { raw = ''; }

  var data;
  try { data = JSON.parse(raw); } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({error: 'Invalid JSON: ' + raw}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(data.sheet);
  if (!sheet) return ContentService.createTextOutput(JSON.stringify({error:'Sheet not found: ' + data.sheet}))
    .setMimeType(ContentService.MimeType.JSON);

  if (data.action === 'append') {
    sheet.appendRow(data.row);
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (data.action === 'update') {
    var rows = sheet.getDataRange().getValues();
    var headers = rows[0];
    var caseIdCol = headers.indexOf('Case ID');
    var statusCol = headers.indexOf('Status') + 1;
    var layerCol  = headers.indexOf('Layer')  + 1;
    var notesCol  = headers.indexOf('Notes')  + 1;
    var updCol    = headers.indexOf('Updated At') + 1;
    for (var i = 1; i < rows.length; i++) {
      if (String(rows[i][caseIdCol]) === String(data.caseId)) {
        if (statusCol) sheet.getRange(i+1, statusCol).setValue(data.status);
        if (layerCol)  sheet.getRange(i+1, layerCol).setValue(data.layer);
        if (notesCol && data.note) sheet.getRange(i+1, notesCol).setValue(data.note);
        if (updCol)    sheet.getRange(i+1, updCol).setValue(data.updatedAt);
        return ContentService.createTextOutput(JSON.stringify({ok:true}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
    return ContentService.createTextOutput(JSON.stringify({error:'row not found for caseId: ' + data.caseId}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({error:'unknown action'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var data = { mcb: [], pr: [] };

  var mcbSheet = ss.getSheetByName('MCB Tracker');
  if (mcbSheet) {
    var rows = mcbSheet.getDataRange().getValues();
    var headers = rows[0];
    var reqCol = headers.indexOf('Requestor?');
    for (var i = 1; i < rows.length; i++) {
      if (!rows[i][reqCol] || String(rows[i][reqCol]).trim() === '') continue;
      var obj = {};
      headers.forEach(function(h, j) {
        var val = rows[i][j];
        if (val instanceof Date) {
          obj[h] = Utilities.formatDate(val, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm:ss');
        } else {
          obj[h] = val;
        }
      });
      data.mcb.push(obj);
    }
  }

  var prSheet = ss.getSheetByName('PR Triggering Submission');
  if (prSheet) {
    var rows2 = prSheet.getDataRange().getValues();
    var headers2 = rows2[0];
    var reqCol2 = headers2.indexOf('Requestor?');
    for (var i = 1; i < rows2.length; i++) {
      if (!rows2[i][reqCol2] || String(rows2[i][reqCol2]).trim() === '') continue;
      var obj2 = {};
      headers2.forEach(function(h, j) {
        var val = rows2[i][j];
        if (val instanceof Date) {
          obj2[h] = Utilities.formatDate(val, Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm:ss');
        } else {
          obj2[h] = val;
        }
      });
      // Inject row index as submission date fallback (row 2 = index 1, etc.)
      if (!obj2['Submitted At'] || String(obj2['Submitted At']).trim() === '') {
        obj2['_rowIndex'] = i;
      }
      data.pr.push(obj2);
    }
  }

  var json = JSON.stringify(data);
  var callback = e && e.parameter && e.parameter.callback;
  if (callback) {
    return ContentService.createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
